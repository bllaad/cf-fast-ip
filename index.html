function ping(ip){ return new Promise(resolve => { let start = performance.now(); const img = new Image(); let host = ip.split(":")[0]; img.onload = img.onerror = () => { resolve(Math.round(performance.now() - start)); }; img.src = "https://" + host + "/favicon.ico?x=" + Date.now(); setTimeout(() => resolve(9999), 3000); }); }
